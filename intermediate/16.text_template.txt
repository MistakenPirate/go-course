package main

import (
	"html/template"
	"os"
)

func main() {
	tmpl := template.New("example")
	tmpl, err := tmpl.Parse("Welcome, {{.name}}! How are you doing?")
	if err != nil {
		panic(err)
	}

	// define data for the welcome message
	data := map[string]interface{}{
		"name": "John",
	}

	err = tmpl.Execute(os.Stdout, data)
	if err != nil {
		panic(err)
	}

}
